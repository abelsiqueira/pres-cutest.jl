\myframe{
  \ctr{CUTEst}
  \begin{itemize}
    \item Repository of Nonlinear Optimization problems;
    \item Provides subroutines to obtain the problem informations;
    \item Decodes the problem, compiles your code with the problem's and run
      your main code;
    \item Widely used.
  \end{itemize}
}

\begin{frame}[fragile]
  \ctr{CUTEst}
\begin{lstlisting}

\end{lstlisting}
\end{frame}
